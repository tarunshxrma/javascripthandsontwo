Question 7 : Find the maximum out of three numbers

Answer:

const Max_out_of_three = (A,B,C) => {
if(A>B && A>C){
return A;
}
if(B>A && B>C){
return B;
}
if(C>A && C>B){
return C;
}
else{
return -1;
}
};


Question 8 : Second Smallest

Answer:

const findSndSmallest = (x,y,z) => {
if((x < y && y < z)||(y < x && y > z)){
return y;
}
else if((y < x && x < z)||(x < y&&x > z)){
return x;
}
else{
return z;
}
};
